# 使用安装 Axios

参考地址: csdn 文章

1.安装 Axios: 使用npm 或 yarn 安装axios到项目中

// 使用npm 安装
npm install axios
//或者 使用yarn 安装
yarn add axios

2.引入 Axios

局部引入 Axios : 的使用非常简单，如果只想在单个组件中使用axios，只需要在这个文件中引入axios，然后就可以直接使用了

import axios from 'axios'
onMounted(() => {
    getData()
})
const getData = async ()=>{
  const response = await axios.get('/hangzhou.json')
  console.log(response.data.features);  
}

**全局引入 Axios **: 如果想要全局引入axios，我们可以借助 vue 提供的 provide 和 inject 方法来实现

在main.js中使用 provide 方法 注入axios，代码如下

import { createApp } from 'vue'
import App from './App.vue'
import router from './router'
import ElementPlus from 'element-plus';
// 引入 axios
import axios from 'axios';

const app = createApp(App);
app.use(router);
app.provide('$axios',axios);
app.mount("#app");

# Axios 封装

在实际的开发中，我们都会将axios进行封装；我在实际的开发中会将网络相关的业务独立放到一个文件夹中，创建两个文件，一个是request.js文件用于封装 axios 请求；一个是api.js文件用于封装所有的 API 请求。

在src文件夹下新建http文件夹，在http文件夹新建request.js文件和api.js文件

文件一: 在request.js中引入axios，并封装 axios 请求，代码如下：

  • 在请求拦截器中添加前置处理，将请求头中添加了一个 Authorization 参数，用于后端 token 权限控制。
  • 在响应拦截器中添加了后置处理，对返回结果进行解析和处理，如果返回结果成功（code 为 200）则返回处理后的数据，否则返回处理后的错误信息

# 请求拦截器 基础版

import axios from 'axios'

const service = axios.create({
    baseURL: process.env.VUE_APP_BASE_API,
    timeout: 5000
})

// 请求拦截器: 希望axios每次发送后台请求都带上token
service.interceptors.request.use(
    config => {
        // 添加请求头等前置处理
        config.headers.token = window.localStorage.getItem('token');
        return config
    },
    error => {
        // 请求错误处理
        console.log('Request Error:', error)
        return Promise.reject(error)
    }
)

// 请求响应拦截器
service.interceptors.response.use(
    response => {
        // 响应后处理
        if (response.status === 200 && response.data.status === 200) {
            return response;
        } else if(response.data.status === 2002 || response.data.status === 502) {
            console.log(response, 'sadad');
            // 显示提示消息
            ElMessage({
                message: '登录已过期，请重新登录',
                type: 'warning',
            });
            // 跳转到登录页面
            router.push('/login');
            // 返回一个带有错误信息的Promise，以阻止进一步的链式调用
            return Promise.reject(new Error('登录已过期，请重新登录...'));
            // 放行（不做任何处理）
            return response;
        } else {
            ElNotification({title: '错误', message: response.data.msg, type: 'error',})
        }
    },
    error => {
        let err = error.response.data;
        let mgs = err.path + '(' + err.status + '): ' + err.message;
        ElNotification({title: '异常', message: mgs, type: 'info',})
        return Promise.reject(error);
    }
);

# 请求拦截器 新版

import axios from 'axios'
import {ElNotification} from "element-plus";
import router from '@/router';
import {getSysLanguage} from "@/util/system";
import {i18n} from '@/main.js';


const service = axios.create({
    baseURL: process.env.VUE_APP_BASE_API,
    timeout: 10000 // 请求的超时时间
})

// 获取系统语言 - 语言切换 zh-CN_中文 en-US_英文 默认英文
const language = getSysLanguage();
// const titleStr = language === 'zh-CN' ? '提示' : 'Prompt';

// 请求拦截器: 希望axios每次发送后台请求都带上token
service.interceptors.request.use(
    config => {
        // 添加请求头等前置处理
        const url = config.url.toLowerCase();
        // 如果请求的 URL 包含 "Clientele"，则不携带 token
        if (!url.includes('clientele')) {
            config.headers.token = window.localStorage.getItem('token');
        }
        // 添加语言请求头 - 语言切换 - zh-CN_中文 en-US_英文 默认英文
        config.headers.language = language;
        return config;
    },
    error => {
        // 请求错误处理
        console.log('Request Error:', error)
        return Promise.reject(error)
    }
)

// 请求响应拦截器
service.interceptors.response.use(
    response => {
        console.log('response:', response)
        // 响应为 下载表格 / 下载文件 - 直接返回
        if (response.headers['content-type'] == 'application/vnd.ms-excel;charset=utf-8'
            || response.headers['content-type'] == 'application/octet-stream'
        ) {
            return response;
        }
        // 响应后处理
        if (response.status === 200 && response.data.status === 200) {
            return response;
        } else if(response.data.status === 2002 || response.data.status === 502 || response.data.status === 2011) {
            // 显示提示消息
            // ElMessage({message: response.data.msg, type: 'warning',});
            ElNotification({title: i18n.global.t('common.dialog.title'), message: response.data.msg, type: 'warning',})
            // 清除 token
            window.localStorage.removeItem('token');
            // 跳转到登录页面
            router.push('/login');
            // 返回一个带有错误信息的Promise，以阻止进一步的链式调用
            return Promise.reject(new Error(response.data.msg));
            // 放行（不做任何处理）
            return response;
        } else {
            ElNotification({title: i18n.global.t('common.dialog.title'), message: response.data.msg, type: 'error',})
        }
    },
    error => {
        let err = error.response.data;
        let mgs = err.path + '(' + err.status + '): ' + err.message;
        ElNotification({title: i18n.global.t('common.dialog.title'), message: err.message, type: 'info',})
        return Promise.reject(error);
    }
);

export default service

# 下载文件类请求

  • 使用默认的 Ajax 请求去下载文件是无法让浏览器自动下载的
  • 必须获取到文件流之后 做成a链接进行下载
  • 必须注意 请求拦截器 记得响应放行 / 下载请求响应的是流并非状态码会被拦截

# Post 请求 + json对象

// 导出表格 data:查询条件对象
export function exportExcel(data) {
    return request({
        url: '/api/spuInfo/exportExcel',
        method: 'post',
        data: JSON.stringify(data), // 将数据转换为 JSON 字符串
        headers: {
            'Content-Type': 'application/json' // 指定内容类型为 JSON
        },
        responseType: 'blob'
    }).then(res => {
        // 文件名前缀 + 时间戳 + 文件后缀
        let fileName = '产品表-' + new Date().getTime() + '.xlsx'
        let blob = new Blob([res.data], {type: "application/vnd.ms-excel;charset=UTF-8"});
        if (window.navigator && window.navigator.msSaveOrOpenBlob) { // IE
            window.navigator.msSaveOrOpenBlob(blob, fileName)
        } else {
            let objectUrl = (window.URL || window.webkitURL).createObjectURL(blob)
            let downFile = document.createElement('a')
            downFile.style.display = 'none'
            downFile.href = objectUrl
            downFile.download = fileName // 下载后文件名
            document.body.appendChild(downFile)
            downFile.click()
            document.body.removeChild(downFile) // 下载完成移除元素
            window.URL.revokeObjectURL(objectUrl)   // 只要映射存在，Blob就不能进行垃圾回收，因此一旦不再需要引用，就必须小心撤销URL，释放掉blob对象。
        }
    }).catch(err=>{
        console.log(err)
    })
}

 * 通过表格流数据并导出Excel

 * @param name 文件名前缀

export function setExportExcel(res, name) {

    let suffix = res.headers['content-disposition'].split('.')[1]
    let fileName = name + new Date().getTime() + '.' + suffix;
