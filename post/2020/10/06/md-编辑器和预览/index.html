# MD 编辑器和预览

# 介绍

  • v-md-editor 是一个基于Vue开发的Markdown编辑器组件
  • v-md-editor: 参考文档

# 依赖安装

# 安装支持vue3的版本

# use npm
npm i @kangc/v-md-editor@next -S

# use yarn
yarn add @kangc/v-md-editor@next

# 在vue3中注册

import { createApp } from 'vue';
import VueMarkdownEditor from '@kangc/v-md-editor';
import '@kangc/v-md-editor/lib/style/base-editor.css';
import vuepressTheme from '@kangc/v-md-editor/lib/theme/vuepress.js';
import '@kangc/v-md-editor/lib/theme/style/vuepress.css';

import Prism from 'prismjs'; // 可能需要提示安装一下

VueMarkdownEditor.use(vuepressTheme, {
  Prism,
});

const app = createApp(/*...*/);

app.use(VueMarkdownEditor);

<template>
  <div id="style-operate">

<script>
import MdEditorView from "@/views/home/MdEditorView";

export default {
  name: "style-operate",
  components: {MdEditorView},
  // data: 对象就是要渲染到页面上的数据
  data: function () {
    return {
    }
  },

  // moubted: 页面加载完毕后执行
  mounted() {

  },

  // methods: 对象中的方法，可以在页面中调用
  methods: {
    // 打开组件编辑器
    getMd(type) {
      this.$refs.editor.dialogFormVisible = true;
      this.$refs.editor.type = type;
      if (type === 1) {
        this.$refs.editor.text = this.form.contentZh;
      } else if (type === 2) {
        this.$refs.editor.text = this.form.contentEn;
      }
    },
    // 编辑器的回调
    callBack(item) {
      if (item.type === 1) {
        this.form.contentZh = item.text;
      } else if (item.type === 2) {
        this.form.contentEn = item.text;
      }
    },
}
</script>

# 编辑器组件

<template>
  <div id="MdEditor">
    <el-dialog v-model="dialogFormVisible" :before-close="beforeClose" title="编辑器" width="1200" draggable>
      <v-md-editor v-model="text" height="600px"></v-md-editor>

      <template #footer>
      <span class="dialog-footer">
        <!-- Form 按钮 -->
        <el-button @click="closeButton()">
          {{ $t('common.button.cancel') }}
        </el-button>
        <el-button type="primary" v-show="type != 'detail'" @click="submitButton()">
          {{ $t('common.button.confirm') }}
        </el-button>
      </span>
      </template>
    </el-dialog>
  </div>
</template>

<script>

export default {

  name: "MdEditor",
  // data: 对象就是要渲染到页面上的数据
  data: function () {
    return {
      // 控制对话框大开闭
      dialogFormVisible: false,

      type: 0,
      text: '',
    }
  },

  // moubted: 页面加载完毕后执行
  mounted() {

  },

  // methods: 对象中的方法，可以在页面中调用
  methods: {
    // 点击遮罩和x时执行
    beforeClose(done) {
      this.closeButton();
      done(); // 必须调用 done，以继续关闭对话框
    },
    // 取消按钮
    closeButton() {
      this.dialogFormVisible= false;
      this.type = 0;
      this.text = '';
    },
    // 确认按钮
    submitButton() {
      this.dialogFormVisible= false;
      // 执行父级组件的方法
      let item = {
        type: this.type,
        text: this.text
      }
      this.$emit('operateClick', item);
    }
  },
}
</script>

# 编辑器

  • 初始化 获取语言和主题进行实时更新
  • 在 <style scoped> 中使用 /deep/ 可进行覆盖组件标签的样式
  • 在 动态导入样式文件 /deep/ 会失效直接使用类名即可
  • /deep/ 深层次的嵌套选择器 在本组件可生效 在样式文件是不会生效的 动态导入的样式文件在编译时已经加载到DOM中，此时的样式选择器是不会被Vue的scoped样式包裹的
<template>
  <div id="MdEditor">
    <el-dialog v-model="dialogFormVisible" :before-close="beforeClose" title="编辑器" width="1200" draggable>
      <v-md-editor v-model="text" height="500px"></v-md-editor>

      <template #footer>
      <span class="dialog-footer">
        <!-- Form 按钮 -->
        <el-button @click="closeButton()">
          {{ $t('common.button.cancel') }}
        </el-button>
        <el-button type="primary" v-show="type != 'detail'" @click="submitButton()">
          {{ $t('common.button.confirm') }}
        </el-button>
      </span>
      </template>
    </el-dialog>
  </div>
</template>

<script>
// 引入编辑器语言包
import mdEnUS from '@kangc/v-md-editor/lib/lang/en-US';
import mdZhCN from '@kangc/v-md-editor/lib/lang/zh-CN';
import VueMarkdownEditor from "@kangc/v-md-editor";
import {getSysLanguage} from "@/util/system";
export default {

  name: "MdEditor",
  // data: 对象就是要渲染到页面上的数据
  data: function () {
    return {
      // 控制对话框大开闭
      dialogFormVisible: false,

      type: 0,
      text: '',
    }
  },

  // moubted: 页面加载完毕后执行
  mounted() {
    // 语言切换
    let language = getSysLanguage();
    if (language == 'en-US') {
      VueMarkdownEditor.lang.use('en-US', mdEnUS);
    } else if (language == 'zh-CN'){
      VueMarkdownEditor.lang.use('zh-CN', mdZhCN);
    }
    // 从缓存中获取主题
    let themeSwitch = window.localStorage.getItem("themeSwitch");
    if (themeSwitch === 'true') {
      // 导入样式文件
      import("@/style/md-dark.css");
    }
  },

  // methods: 对象中的方法，可以在页面中调用
  methods: {
    // 点击遮罩和x时执行
    beforeClose(done) {
      this.closeButton();
      done(); // 必须调用 done，以继续关闭对话框
    },
    // 取消按钮
    closeButton() {
      this.dialogFormVisible= false;
      this.type = 0;
      this.text = '';
    },
    // 确认按钮
    submitButton() {
      this.dialogFormVisible= false;
      // 执行父级组件的方法
      let item = {
        type: this.type,
        text: this.text
      }
      this.$emit('operateClick', item);
    }
  },
}
</script>

# 国际化

# 介绍

v-md-editor 默认使用中文作为语言。如果需要使用其他语言，可以参考以下方案。

# 切换语言

可以通过use方法切换当前使用的语言。

import enUS from '@kangc/v-md-editor/lib/lang/en-US';

VueMarkdownEditor.lang.use('en-US', enUS);

按钮方法修改

// 导入语言文件
import mdEnUS from '@kangc/v-md-editor/lib/lang/en-US';
import mdZhCN from '@kangc/v-md-editor/lib/lang/zh-CN';
import VueMarkdownEditor from "@kangc/v-md-editor";

// 在组件初始化时进行执行 注意改变了语言但是刷新会重置为默认中文所以需要初始化执行
let language = getSysLanguage(); // 获取目前的国际化语言
if (language == 'en-US') {
  VueMarkdownEditor.lang.use('en-US', mdEnUS); // 设置编辑器的语言文英文
} else if (language == 'zh-CN'){
  VueMarkdownEditor.lang.use('zh-CN', mdZhCN); // 设置编辑器的语言文中文
}

# 修改默认配置

可以使用复制方法来修改和扩展副本。

VueMarkdownEditor.lang.add({
  'zh-CN': {
    h1: {
      toolbar: 'title 1',
    },
  },
});
