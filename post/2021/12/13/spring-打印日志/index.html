Spring 打印日志 可在控制台打印请求的详情

# Spring 拦截器 打印接口

# 1.自定义拦截器

一.实现 HandlerInterceptor 拦截器接口 重写三个方法

package com.apai.config.Interceptor;

import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.ModelAndView;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.Enumeration;
import java.util.Map;


/**
 * @Author Jun: 第一步是实现HandlerInterceptor接口来定义一个拦截器，第二步就是配置拦截器了
 * @Date 2023/07/19
 * @Description: 日志拦截器
 */
public class PortLogInterceptor implements HandlerInterceptor {

    /**
     * preHandle方法：
     * 1. 叫做预处理方法，本方法在控制器方法（Controller的方法）之前执行
     * 2. 用户的请求最先到达此方法，在这个方法中可以获取请求的信息，验证请求是否符合要求
     * 3. 以验证用户是否登录，验证用户是否有权限访问某个链接地址（url）。如果返回true则放行，返回false则拦截。
     * @param request 请求对象
     * @param response 响应对象
     * @param handler 处理器对象
     * @return true 放行，执行下一个拦截器，如果没有则执行controller中的方法 false 拦截，不执行下一个拦截器，也不执行controller中的方法
     * @throws Exception
     */
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {

        // 获取请求的url /role/listRole
        // String servletPath = request.getServletPath();
        // 获取请求的url http://localhost:8003/role/listRole
        StringBuffer requestURL = request.getRequestURL();
        // 获取请求的方式
        String method = request.getMethod();
        // 获取请求的时间
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        // 获取请求头中的 token
        String token = request.getHeader("token");
        System.out.println("||- 接口请求时间: --> " + simpleDateFormat.format(Calendar.getInstance().getTime()));
        System.out.println("接口请求路径: --> " + requestURL);
        System.out.println("接口请求方式: --> " + method);
        System.out.println("接口请求头中的 token: --> " + token);
        // true 放行，执行下一个拦截器，如果没有则执行controller中的方法 false 拦截，不执行下一个拦截器，也不执行controller中的方法
        return true;
    }

    /**
     * postHandle方法：
     * 1. 叫做后处理方法。在controller中的方法之后执行的
     * 2. 能够获取到处理器方法的返回值 mv，可以修改mv中的数据和视图
     * 3. 可以影响到最后的执行结果。主要是对原来的执行结果做二次修正
     */
    @Override
    public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {
        HandlerInterceptor.super.postHandle(request, response, handler, modelAndView);
    }

    /**
     * afterCompletion方法：
     * 1. 最后执行的方法，在页面渲染之后执行
     * 2. 在请求处理完成后执行的，框架中规定是当你的视图处理完成后，对视图进行了forword就任务请求处理完成
     * 3. 一般做资源回收工作的，程序请求过程中创建了一些对象。在这里可以删除，吧占用的内存回收
     */
    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response, Object handler, Exception ex) throws Exception {
        HandlerInterceptor.super.afterCompletion(request, response, handler, ex);
    }
}

# 2.配置拦截器

配置自定义拦截器 选择拦截方式

package com.apai.config.Interceptor;

import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

/**
 * 拦截器配置类
 */
@Configuration
public class MyInterceptorConfig implements WebMvcConfigurer {

    /**
     * 配置拦截器
     *
     * @param registry
     */
    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        // 选择自定义拦截器
        registry.addInterceptor(new PortLogInterceptor())
                // 需要拦截的请求
                .addPathPatterns("/api/lsl/**")
                .addPathPatterns("/**")
                // 不拦截的请求
                .excludePathPatterns("/api/debug/**");
    }
}
